Question -1 : Statement to create the Contact table
**Answer:**
create table tbl_contact
(
    contactID int primary key,
    companyID int,
    FirstName varchar(45),
    LastName Varchar(45),
    street varchar(45),
    city varchar(45),
    state Varchar(2),
    zip varchar(10),
    Ismain boolean,
    Email varchar(45),
    phone varchar(12),
    companyID int REFERENCES company(companyID)
)
